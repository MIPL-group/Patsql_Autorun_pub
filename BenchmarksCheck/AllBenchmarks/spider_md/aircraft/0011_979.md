FROM aircraft AS T1
JOIN airport_aircraft AS T2 ON T1.Aircraft_ID = T2.Aircraft_ID
JOIN airport AS T3 ON T2.Airport_ID = T3.Airport_ID
WHERE T3.Total_Passengers > 10000000
