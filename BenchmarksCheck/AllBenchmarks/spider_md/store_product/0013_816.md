SELECT t3.headquartered_city
FROM store AS t1
JOIN store_district AS t2 ON t1.store_id = t2.store_id
JOIN district AS t3 ON t2.district_id = t3.district_id
GROUP BY t3.headquartered_city
ORDER BY count(*) DESC
LIMIT 1
