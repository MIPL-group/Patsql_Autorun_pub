SELECT T1.id,
       T1.name
FROM enzyme AS T1
JOIN medicine_enzyme_interaction AS T2 ON T1.id = T2.enzyme_id
WHERE T2.interaction_type = 'activitor'
GROUP BY T1.id
ORDER BY count(*) DESC
LIMIT 1
