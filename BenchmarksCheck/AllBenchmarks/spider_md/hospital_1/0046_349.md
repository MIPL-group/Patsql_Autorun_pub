```sql
SELECT T1.name
FROM physician AS T1
JOIN affiliated_with AS T2 ON T1.EmployeeID = T2.physician
JOIN department AS T3 ON T2.department = T3.DepartmentID
WHERE T3.name = 'Surgery'
  OR T3.name = 'Psychiatry'
```
