SELECT COUNT(*)
FROM
  (SELECT T1.state
   FROM college AS T1
   JOIN tryout AS T2 ON T1.cName = T2.cName
   WHERE T2.pPos = 'mid'
   EXCEPT SELECT T1.state
   FROM college AS T1
   JOIN tryout AS T2 ON T1.cName = T2.cName
   WHERE T2.pPos = 'goalie')
