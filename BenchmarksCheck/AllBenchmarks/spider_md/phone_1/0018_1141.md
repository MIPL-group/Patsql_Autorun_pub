SELECT DISTINCT T2.Hardware_Model_name
FROM screen_mode AS T1
JOIN phone AS T2 ON T1.Graphics_mode = T2.screen_mode
WHERE t2.Company_name = "Nokia Corporation"
  AND T1.Type != "Text";
