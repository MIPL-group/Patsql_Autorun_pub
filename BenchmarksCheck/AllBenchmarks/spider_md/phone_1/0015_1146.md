SELECT T1.WiFi,
       T3.Type
FROM chip_model AS T1
JOIN phone AS T2 ON T1.Model_name = T2.chip_model
JOIN screen_mode AS T3 ON T2.screen_mode = T3.Graphics_mode
WHERE T2.Hardware_Model_name = "LG-P760";
