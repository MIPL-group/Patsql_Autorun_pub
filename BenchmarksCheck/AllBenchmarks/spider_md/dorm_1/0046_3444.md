# solution

SELECT T1.dorm_name,
       T1.student_capacity
FROM dorm AS T1
JOIN has_amenity AS T2 ON T1.dormid = T2.dormid
JOIN dorm_amenity AS T3 ON T2.amenid = T3.amenid
GROUP BY T2.dormid
ORDER BY count(*)
LIMIT 1
