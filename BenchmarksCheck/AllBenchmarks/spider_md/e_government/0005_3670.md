SELECT t1.party_email
FROM parties AS t1
JOIN party_forms AS t2 ON t1.party_id = t2.party_id
WHERE t2.form_id =
    (SELECT form_id
     FROM party_forms
     GROUP BY form_id
     ORDER BY count(*) DESC
     LIMIT 1)
