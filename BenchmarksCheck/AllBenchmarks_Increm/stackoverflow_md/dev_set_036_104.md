# solution

```sql
SELECT
    T3.T0_PID,
    T3.T0_GUID,
    T2.LastName,
    T2.FirstName 
FROM
    input2 AS T2 
JOIN
    (
        SELECT
            T0.GUID AS T0_GUID,
            T0.PID AS T0_PID,
            count(DISTINCT T1.GUID) AS count_distinct_T1_GUID 
        FROM
            input1 AS T0 
        JOIN
            input1 AS T1 
                ON T0.PID = T1.PID 
        GROUP BY
            T0.GUID,
            T0.PID
    ) AS T3 
        ON T3.T0_GUID = T2.GUID 
WHERE
    T3.count_distinct_T1_GUID >= 2 
ORDER BY
    T3.T0_PID ASC
```
