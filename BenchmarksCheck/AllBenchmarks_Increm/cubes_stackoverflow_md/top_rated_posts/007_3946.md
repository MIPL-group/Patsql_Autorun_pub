SELECT tt.*
FROM i007 AS tt
JOIN
    (SELECT home, MAX(datetime) AS MaxDateTime

    GROUP BY home) groupedtt
ON tt.home = groupedtt.home
AND tt.datetime = groupedtt.MaxDateTime
